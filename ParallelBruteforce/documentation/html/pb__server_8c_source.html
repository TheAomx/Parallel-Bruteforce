16 static void createClientTasks(int numTasks, long numPasswords, char* startPW, char* endPw, ClientTask* out, PwGenAlgoType pwGenAlgoType) {
17  int isDivider = ((numPasswords % numTasks) == 0) ? 1 : 0;
18  ulong passwordsPerTask = numPasswords / numTasks;
19  ulong currentOffset = toNumberIndefaultAlphabet(startPW);
20 
21  /*Calculate the number of tasks which are all calculating the same amount of work.*/
22  int fullLengthTasks = (isDivider == 1) ? (numTasks) : (numTasks - 1);
23  char* tempPwd;
24  char* tempPwd2;
25 
26  /*Initialize the first 'fullLengthTasks' tasks with an equal amount of work...*/
27  for (int i = 0; i < fullLengthTasks; i++) {
28  /*Temp Pointers for start and end*/
29  tempPwd = (char*) malloc(sizeof (char)*MAX_PASSWORD);
30  tempPwd2 = (char*) malloc(sizeof (char)*MAX_PASSWORD);
31  memset(tempPwd, '\0', sizeof (char)*MAX_PASSWORD);
32  memset(tempPwd2, '\0', sizeof (char)*MAX_PASSWORD);
33 
34  getPasswordAt(currentOffset, tempPwd);
35  getPasswordAt(passwordsPerTask + currentOffset, tempPwd2);
36 
37  currentOffset += passwordsPerTask;
38 
39  initClientTask((out + i), passwordsPerTask, tempPwd, tempPwd2, pwGenAlgoType);
40  }
41  /* If the number of passwords could not be divided into equal parts, the last client has to do a bit more work than others.
42  * The amount of work to do for this task can be defined as follows: work_this = work_all - (work_others*num_others)
43  */
44 
45  if (isDivider == 0) {
46  /*Temp Pointers for start and end*/
47  tempPwd = (char*) malloc(sizeof (char)*MAX_PASSWORD);

50  memset(tempPwd2, '\0', sizeof (char)*MAX_PASSWORD);

54  getPasswordAt(currentOffset, tempPwd);
