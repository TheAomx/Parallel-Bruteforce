132 /*
133  * One round of MD5. The data must be aligned for 32-bit access.
134  */
135 static void
136 md5_round(const unsigned char *data, sph_u32 r[4])
137 {
138  /*
139  * On machines with fast little-endian decoding, we simply
140  * reread from the input buffer.
141  */
142 #if SPH_LITTLE_FAST
143 
144 #define X(idx) sph_dec32le_aligned(data + 4 * (idx))
145 
146 #else
147 
148  sph_u32 X_var[16];
149  int i;
150 
151  for (i = 0; i < 16; i ++)
152  X_var[i] = sph_dec32le_aligned(data + 4 * i);
153 #define X(idx) X_var[idx]
154 
155 #endif
156 
157  MD5_ROUND_BODY(X, r);
158 
159 #undef X
160 }
