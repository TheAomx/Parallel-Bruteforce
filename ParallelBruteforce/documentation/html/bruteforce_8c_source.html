47 static void printProgess(BruteforceProgressStats *stats, int numThreads, ulong currentPasswordIndex, ulong numPasswords, char *currentPassphrase) {
48  const ulong perfCounterCheckIntervall = 10000000;
49 
50  if (stats->takeTimeOffset == 1) {
51  stats->time = getElapsedTime();
52  stats->takeTimeOffset = 0;
53  }
54 
55  if ((currentPasswordIndex % perfCounterCheckIntervall) == 0) {
56  int threadID = getThreadID();
57  ulong needToCheck = (numPasswords / numThreads);
58  ulong checkedPws = currentPasswordIndex - threadID * needToCheck;
59  double percentFinished = (double) checkedPws / (double) needToCheck;
60  percentFinished *= 100.0;
61 
62  stats->timeDiff = getElapsedTime() - stats->time;
63  stats->takeTimeOffset = 1;
64  stats->pwDiff = checkedPws - stats->checkedLast;
65 
66  if (currentPasswordIndex == 0 || stats->pwDiff <= 0) {
67  stats->kiloHashesPerSecond = 0.0;
68  stats->timeDiff = 0.0;
69  }
70  else {
71  stats->kiloHashesPerSecond = (((double) stats->pwDiff) * (1.0 / stats->timeDiff)) / 1000.0;
72 
73  }
74 
75  printf("[%d] %.2f%% %s, %.3f msec(per %ld hashes) -> %.3f kHashes/s.\n", threadID, percentFinished, currentPassphrase, stats->timeDiff * 1000.0, perfCounterCheckIntervall, stats->kiloHashesPerSecond);
76  fflush(stdout);
77 
78  stats->checkedLast = checkedPws;
79  }
80 }
